import numpy as np

def get_combined_expression_v2(
    pt: np.ndarray,
    wkr: np.ndarray,
    rm: np.ndarray,
    so: np.ndarray,
    twk: np.ndarray,
    ema: np.ndarray,
) -> np.ndarray:
    """
    Advanced composite heuristic for Job Shop Scheduling prioritization.

    Enhancements over previous versions:
    - Dynamic EMA clipping bounds adapt to EMA distribution percentiles to preserve data-driven flexibility.
    - EMA gating uses smooth shifted sigmoid with adjustable slope for finer control.
    - Workload urgency blends linear and nonlinear terms for enhanced sensitivity.
    - Lookahead gate softened with scaled softplus to avoid sharp drop-offs.
    - Softplus scaling for total workload with adaptive temperature.
    - Weight factor uses a curved smooth polynomial on pt_n emphasizing mid-range processing times.
    - Final combination blends additive and multiplicative terms to better leverage complementary signals.
    - Global normalization via robust max/mean statistics with epsilon for numerical stability.

    Parameters
    ----------
    pt : np.ndarray
        Processing time of the current operation for all workpieces.
    wkr : np.ndarray
        Remaining processing time for all workpieces.
    rm : np.ndarray
        Remaining processing time excluding current operation.
    so : np.ndarray
        Processing time of the succeeding operation for all workpieces.
    twk : np.ndarray
        Total processing time for all workpieces.
    ema : np.ndarray
        Exponential Moving Average of relative deviations between realized and nominal times.

    Returns
    -------
    np.ndarray
        Composite heuristic priority scores for scheduling decisions.
    """
    eps = 1e-9
    
    # Normalize inputs robustly to max or mean to reduce scale sensitivity
    max_pt = np.max(pt)
    max_wkr = np.max(wkr)
    max_rm = np.max(rm)
    max_so = np.max(so)
    max_twk = np.max(twk)
    max_pt = max_pt if max_pt > eps else 1.0
    max_wkr = max_wkr if max_wkr > eps else 1.0
    max_rm = max_rm if max_rm > eps else 1.0
    max_so = max_so if max_so > eps else 1.0
    max_twk = max_twk if max_twk > eps else 1.0
    
    pt_n = pt / max_pt
    wkr_n = wkr / max_wkr
    rm_n = rm / max_rm
    so_n = so / max_so
    twk_n = twk / max_twk

    # Dynamic asymmetric clipping of EMA based on data percentiles (10th and 90th) for adaptive range
    p10, p90 = np.percentile(ema, [10, 90])
    low_clip = min(-0.3, p10)
    high_clip = max(0.6, p90)
    ema_clip = np.clip(ema, low_clip, high_clip)

    # Shifted sigmoid gating for EMA emphasizing positive deviations with controllable slope & offset
    slope_ema = 8.0
    offset_ema = 0.15
    ema_gate = 1 / (1 + np.exp(-slope_ema * (ema_clip - offset_ema)))

    # Composite workload urgency blending linear and nonlinear terms, smoothed sigmoid centered near 0.5
    workload_linear = wkr_n * (1 + 0.8 * rm_n)
    workload_sig = 1 / (1 + np.exp(-12 * (workload_linear - 0.5)))
    workload = 0.4 * workload_linear + 0.6 * workload_sig  # weighted blend for sensitivity

    # Lookahead gate softened via softplus with temperature; avoids harsh cutoffs on ema positive deviations
    ema_pos = np.maximum(ema_clip - 0.2, 0)
    softplus_temp = 10.0
    lookahead_gate = np.log1p(np.exp(-softplus_temp * ema_pos)) / softplus_temp  # decreasing smooth gate
    lookahead = so_n * lookahead_gate

    # Adaptive softplus scaling for total workload; temp chosen to retain meaningful gradient around mean range
    softplus_temp_twk = 14.0
    softplus_twk = np.log1p(np.exp(twk_n * softplus_temp_twk)) / softplus_temp_twk + eps

    # Smooth polynomial weighting curve on pt emphasizing mid-range values to avoid extremes dominating
    weight_pt = 0.7 + 0.9 * (3 * pt_n**2 - 2 * pt_n**3)  # smoothstep cubic polynomial from 0->1 over pt_n
    weight = (1 + 1.1 * ema_gate) * weight_pt

    # Final combination blending multiplicative urgency/lookahead with additive weighted pt + ema to capture complementary signals
    combined = ((weight * workload * lookahead) + 0.3 * (weight + workload)) / softplus_twk

    return combined
