import numpy as np

def get_combined_expression_v2(
    pt: np.ndarray,
    wkr: np.ndarray,
    rm: np.ndarray,
    so: np.ndarray,
    twk: np.ndarray,
    ema: np.ndarray
) -> np.ndarray:
    """
    Improved composite expression for job shop scheduling heuristic that:
    - Uses tanh for smooth bounded EMA scaling around 1.
    - Applies gentle root transforms to moderate influence of lookahead and remaining work.
    - Incorporates clipping to ensure numerical stability and interpretability.
    - Balances weighted terms for enhanced sensitivity and robustness.
    """
    eps = 1e-9

    # Smooth bounded EMA scaling approx in [0.6, 1.4], centered near 1
    ema_scaled = 1.0 + 0.4 * np.tanh(5 * (ema - 0.1))

    # Lookahead term: fourth root of the positive difference so - pt (gentle sensitivity)
    lookahead = np.power(np.maximum(so - pt, 0), 0.25)

    # Time pressure ratio clipped between 0 and 1, soft penalty for exceeding remaining work over total
    time_pressure = np.clip((wkr - rm) / (twk + eps), 0, 1)

    # Remaining work softened by cube root with offset +1, ensuring smooth gradient and positivity
    remaining_work = np.cbrt(rm + 1)

    # Processing time softened by square root to reduce dominance of large pt values
    pt_soft = np.sqrt(pt + eps)

    # Composite expression balancing terms to improve heuristic effectiveness
    combined = (
        pt_soft * ema_scaled               # weighted processing time adjusted by EMA
        + 0.5 * lookahead                  # softened lookahead influence
        - 1.1 * time_pressure              # moderate penalty on time pressure
        + 0.9 * remaining_work             # smoothed remaining work influence
    )

    return combined
