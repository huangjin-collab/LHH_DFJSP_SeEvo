import numpy as np

def get_combined_expression_v2(
    pt: np.ndarray,
    wkr: np.ndarray,
    rm: np.ndarray,
    so: np.ndarray,
    twk: np.ndarray,
    ema: np.ndarray
) -> np.ndarray:

    eps = 1e-8
    # Normalize all time-based variables by total workload per workpiece to maintain scale balance
    pt_n = pt / (twk + eps)
    wkr_n = wkr / (twk + eps)
    rm_n = rm / (twk + eps)
    so_n = so / (twk + eps)

    # Nonlinear EMA penalties:
    # - Exponential gives a rapidly increasing penalty for variability beyond zero
    # - Tanh smooths penalty with steep slope near zero, saturating to bound variation
    ema_exp = np.clip(np.expm1(2.5 * np.clip(ema, a_min=0, a_max=None)), 0, 6)  # expm1 for (exp(x)-1)
    ema_tanh = np.tanh(4 * ema)  # sharpen tanh slope for variability

    # Lookahead weighted by EMA exponential penalty emphasizing unstable next ops to de-prioritize
    lookahead = so_n * (1 + ema_exp)

    # Urgency favors jobs with less remaining and low variability:
    # Use (1 - wkr_n) to prioritize those nearing completion,
    # dampened strongly by tanh-based EMA penalty to avoid uncertain ops
    urgency = (1 - wkr_n) * (1 - ema_tanh)

    # Immediate cost scaled by processing time and increased by variability penalty (EMA exponential)
    immediate_cost = pt_n * (1 + 0.85 * ema_exp)

    # Remaining work factor discounts rem workload by a strong linear penalty on variability,
    # capped to not go negative to avoid destabilizing priorities
    rem_factor_raw = rm_n * (1 - 0.9 * ema)
    remaining_factor = np.maximum(rem_factor_raw, 0)

    # Compose final combined expression:
    # - Immediate cost is most important (weight 1)
    # - Remaining work discounted but important (weight 0.8)
    # - Lookahead subtracts score to lower priority on risky successors (weight 0.65)
    # - Urgency subtracts to emphasize stable near-completion jobs (weight 1.0)
    combined_expr = (
        immediate_cost
        + 0.8 * remaining_factor
        - 0.65 * lookahead
        - 1.0 * urgency
    )

    return combined_expr
